# Utilise l'image Laravel Sail officielle (PHP 8.3 + Composer + extensions courantes)
FROM laravelsail/php83-composer:latest

# Installer Node.js et npm (pour Vite et le build front)
RUN curl -fsSL https://deb.nodesource.com/setup_20.x | bash - && \
    apt-get update && apt-get install -y nodejs

# S'assurer que l'extension pdo_mysql est installée (pour MySQL)
RUN docker-php-ext-install pdo_mysql

# Définir le dossier de travail
WORKDIR /var/www
COPY . /var/www

# Copier le script d'entrée et le rendre exécutable
COPY entrypoint.sh /entrypoint.sh
RUN chmod +x /entrypoint.sh

# Exposer le port utilisé par Laravel (8000 par défaut)
EXPOSE 8000

# Lancer le script d'entrée
ENTRYPOINT ["/entrypoint.sh"]
